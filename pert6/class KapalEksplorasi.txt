public class KapalEksplorasi extends KendaraanGalaksi {
    private int modulScan;

    public KapalEksplorasi(String nama, int kapasitas, int modulScan) {
        super(nama, kapasitas);
        this.modulScan = modulScan;
    }

    @Override 
    void aktifkanMesin() {
        if (levelEnergi < 15) System.out.println("Energi tidak mencukupi!");
        else System.out.println("Kapal eksplorasi siap berangkat!");
    }

    @Override 
    void jelajah(int jarak) {
        if (levelEnergi >= jarak * 2) {
            levelEnergi -= jarak * 2;
            System.out.println("Kapal eksplorasi menjelajah sejauh " + jarak + " km.");
        } else System.out.println("Energi tidak cukup.");
    }

    @Override 
    void isiEnergi(int jumlah) { levelEnergi = Math.min(100, levelEnergi + jumlah); 
    }

    void scanPlanet(String nama) {
        System.out.println("Melakukan scan pada planet " + nama + " dengan modul level " + modulScan + ".");
    }
}
